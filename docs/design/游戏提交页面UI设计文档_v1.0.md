function parseMarkdown(markdown) {
  // ç®€åŒ–çš„Markdownè§£æ
  // å®é™…é¡¹ç›®ä¸­åº”è¯¥ä½¿ç”¨æˆç†Ÿçš„åº“ï¼ˆå¦‚marked.jsï¼‰
  return markdown
    .replace(/^# (.*$)/gim, '<h1>$1</h1>')
    .replace(/^## (.*$)/gim, '<h2>$1</h2>')
    .replace(/^### (.*$)/gim, '<h3>$1</h3>')
    .replace(/\*\*(.*)\*\*/gim, '<strong>$1</strong>')
    .replace(/\*(.*)\*/gim, '<em>$1</em>')

}

function insertMarkdownSyntax(action) {
  const editor = document.getElementById('content');
  const start = editor.selectionStart;
  const end = editor.selectionEnd;
  const text = editor.value;
  const selectedText = text.substring(start, end);

  let syntax = '';
  switch (action) {
    case 'bold':
      syntax = `**${selectedText || 'ç²—ä½“æ–‡æœ¬'}**`;
      break;
    case 'italic':
      syntax = `*${selectedText || 'æ–œä½“æ–‡æœ¬'}*`;
      break;
    case 'link':
      syntax = `[${selectedText || 'é“¾æ¥æ–‡æœ¬'}](url)`;
      break;
    case 'image':
      syntax = `![${selectedText || 'å›¾ç‰‡æè¿°'}](url)`;
      break;
    case 'code':
      syntax = `\`${selectedText || 'ä»£ç '}\``;
      break;
    case 'quote':
      syntax = `> ${selectedText || 'å¼•ç”¨å†…å®¹'}`;
      break;
  }

  editor.value = text.substring(0, start) + syntax + text.substring(end);
  editor.focus();
}
